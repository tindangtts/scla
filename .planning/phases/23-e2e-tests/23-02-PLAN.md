```typescript
import { test, expect } from '@playwright/test';
import { loginAsResident } from '../helpers/auth';

test.describe('Ticket Creation and Chat', () => {

  test('creates a new ticket, opens detail, sends a chat message', async ({ page }) => {
    await loginAsResident(page);

    // Navigate to Star Assist
    await page.goto('/star-assist');
    await expect(page.getByTestId('button-new-ticket')).toBeVisible();

    // Click new ticket
    await page.getByTestId('button-new-ticket').click();
    await expect(page.getByTestId('input-title')).toBeVisible();

    // Fill in ticket form
    const ticketTitle = `E2E Test Ticket ${Date.now()}`;
    await page.getByTestId('input-title').fill(ticketTitle);

    // Select category — use "general_enquiry" (known category from seed)
    await page.getByTestId('category-general_enquiry').click();

    // Select service type — look for the first available service type radio
    // Service types are rendered as radio buttons with data-testid="service-type-{value}"
    await page.getByTestId('service-type-information-request').click();

    // Fill unit number (resident has unit A-12-03 but it may be pre-filled)
    const unitInput = page.getByTestId('input-unit');
    const unitValue = await unitInput.inputValue();
    if (!unitValue) {
      await unitInput.fill('A-12-03');
    }

    // Fill description
    await page.getByTestId('input-description').fill('This is an automated E2E test ticket for testing the full ticket creation and chat flow.');

    // Submit
    await page.getByTestId('button-submit-ticket').click();

    // After submission, should redirect to star-assist list or ticket detail
    // Wait for navigation to complete — could go to ticket detail or list
    // The app redirects to /star-assist after ticket creation
    await page.waitForURL(/star-assist/, { timeout: 10000 });

    // If we're on the list page, find the new ticket and click it
    // The ticket should appear in the list with our title
    if (page.url().match(/\/star-assist$/)) {
      // Look for the ticket card with our title text
      const ticketCard = page.getByText(ticketTitle).first();
      await expect(ticketCard).toBeVisible({ timeout: 10000 });
      await ticketCard.click();
    }

    // Now on ticket detail page — verify status badge is visible
    await expect(page.getByTestId('text-ticket-status')).toBeVisible({ timeout: 10000 });

    // Send a chat message
    const chatMessage = `E2E chat message ${Date.now()}`;
    await page.getByTestId('input-chat-message').fill(chatMessage);
    await page.getByTestId('button-send-chat').click();

    // Assert the message appears in the chat thread
    await expect(page.getByText(chatMessage)).toBeVisible({ timeout: 10000 });
  });

});
```

Key design decisions:
- Uses unique ticket title with timestamp to avoid collision with seeded data.
- Selects "general_enquiry" category (known from seed data categories).
- After submission, handles both possible redirects (list or detail).
- Sends a chat message and verifies it appears in the thread.
- Uses generous timeouts (10s) for network round-trips.

```typescript
import { test, expect } from '@playwright/test';
import { loginAsResident } from '../helpers/auth';

test.describe('Facility Booking and Cancellation', () => {

  test('books a facility slot and cancels it', async ({ page }) => {
    await loginAsResident(page);

    // Navigate to bookings page
    await page.goto('/bookings');

    // The bookings page has tabs: "Book" (facilities list), "Upcoming", "Completed", "Cancelled"
    // Default tab should show facilities for booking — click "Book" tab explicitly
    await page.getByTestId('tab-book').click();

    // Select a facility — use the first available facility card
    // Seeded facilities: Olympic Swimming Pool, Tennis Court A, etc.
    const facilityCard = page.locator('[data-testid^="card-facility-"]').first();
    await expect(facilityCard).toBeVisible({ timeout: 10000 });
    await facilityCard.click();

    // Now on booking detail page — select a date
    // Pick a future date. The date buttons have data-testid="date-{YYYY-MM-DD}"
    // We need to find an available date button and click it
    // Wait for date buttons to render
    const dateButton = page.locator('[data-testid^="date-"]').first();
    await expect(dateButton).toBeVisible({ timeout: 10000 });
    await dateButton.click();

    // Select a time slot
    // Wait for slots to load after date selection
    const slotButton = page.locator('[data-testid^="slot-"]').first();
    await expect(slotButton).toBeVisible({ timeout: 10000 });
    await slotButton.click();

    // Confirm booking
    const confirmButton = page.getByTestId('button-confirm-booking');
    await expect(confirmButton).toBeEnabled({ timeout: 5000 });
    await confirmButton.click();

    // After successful booking, should show a success state with "View Bookings" button
    // or redirect to bookings list
    await expect(page.getByTestId('button-view-bookings')).toBeVisible({ timeout: 10000 });
    await page.getByTestId('button-view-bookings').click();

    // Now on bookings list — switch to "Upcoming" tab
    await page.getByTestId('tab-upcoming').click();

    // Verify the new booking appears in the list
    // There should be at least one booking card visible (the one we just created + seeded BK-0001)
    const bookingCard = page.locator('[data-testid^="card-booking-"]').first();
    await expect(bookingCard).toBeVisible({ timeout: 10000 });

    // Cancel the first upcoming booking
    // The cancel button is inside a button group on the booking card
    const cancelButton = page.locator('[data-testid^="button-cancel-group-"]').first();
    await expect(cancelButton).toBeVisible();
    await cancelButton.click();

    // A confirmation dialog may appear — look for a confirm button in the dialog
    // Most UI patterns use AlertDialog with a confirm/cancel pair
    const dialogConfirm = page.getByRole('button', { name: /confirm|yes|cancel booking/i });
    if (await dialogConfirm.isVisible({ timeout: 3000 }).catch(() => false)) {
      await dialogConfirm.click();
    }

    // After cancellation, the booking should move to the "Cancelled" tab
    // Switch to Cancelled tab and verify
    await page.getByTestId('tab-cancelled').click();

    // There should be at least one cancelled booking (the one we just cancelled + seeded BK-0003)
    const cancelledCard = page.locator('[data-testid^="card-booking-"]').first();
    await expect(cancelledCard).toBeVisible({ timeout: 10000 });
  });

});
```

Key design decisions:
- Uses locator patterns (`[data-testid^="card-facility-"]`) instead of specific IDs since facility IDs are auto-generated.
- Selects first available date and first available time slot (flexible against different seeded dates).
- Handles potential confirmation dialog for cancellation.
- Verifies booking appears in "Upcoming" tab, then after cancel appears in "Cancelled" tab.
- Uses generous timeouts for API round-trips and rendering.

IMPORTANT: When running this test, carefully observe the actual UI behavior. The exact flow may differ from what's described in the data-testids. Adjust selectors based on the actual rendered DOM. Key things to watch:
