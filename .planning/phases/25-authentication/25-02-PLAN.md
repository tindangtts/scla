export const upgradeRequestsTable = pgTable("upgrade_requests", {

  userId: uuid("user_id").notNull().references(() => usersTable.id),
  userName: text("user_name").notNull(),
  userEmail: text("user_email").notNull(),
  unitNumber: text("unit_number").notNull(),
  residentId: text("resident_id").notNull(),
  developmentName: text("development_name").notNull(),

  submittedAt: timestamp("submitted_at").notNull().defaultNow(),
  reviewedAt: timestamp("reviewed_at"),

});

2. **Create `artifacts/web/src/app/(resident)/upgrade/actions.ts`** — Server Action:
   - `"use server"` directive.
   - Export `async function submitUpgradeRequest(prevState: { error?: string; success?: boolean }, formData: FormData)`.
   - Get current user via `createClient()` then `supabase.auth.getUser()`. If no user, return `{ error: "Not authenticated" }`.
   - Query DB user by email to get the `users.id`.
   - Validate: unitNumber, residentId, developmentName are non-empty.
   - Check user doesn't already have a pending request: `db.select().from(upgradeRequestsTable).where(and(eq(upgradeRequestsTable.userId, dbUser.id), eq(upgradeRequestsTable.status, 'pending')))`. If exists, return `{ error: "You already have a pending request" }`.
   - Insert into `upgradeRequestsTable`: `{ userId: dbUser.id, userName: dbUser.name, userEmail: dbUser.email, unitNumber, residentId, developmentName, status: 'pending' }`.
   - Update `usersTable` set `upgradeStatus = 'pending'` where id = dbUser.id.
   - Return `{ success: true }`.
   - Call `revalidatePath('/upgrade')`.

   - Async Server Component. Import `requireAdmin` from `@/lib/auth` and call it (handles redirect if not admin).

   - Query all upgrade requests: `db.select().from(upgradeRequestsTable).orderBy(desc(upgradeRequestsTable.submittedAt))`.
   - Render a table/list of requests showing: userName, userEmail, unitNumber, residentId, developmentName, status, submittedAt.
   - For "pending" requests, show Approve and Reject buttons. These are forms that call `approveUpgrade` and `rejectUpgrade` Server Actions with the request ID as a hidden field.

   - `"use server"` directive.
   - Export `async function approveUpgrade(formData: FormData)`:
     - Get requestId from formData.
     - Get current admin user via `createClient()` + `getUser()` (verify they're staff via Drizzle query).
     - Get the upgrade request by ID. If not found or not "pending", return early.
     - **In a transaction-like sequence** (sequential queries, not pg transaction since Drizzle serverless):
       a. Update `upgradeRequestsTable` set `status = 'approved'`, `reviewedAt = new Date()`.

       c. Update Supabase Auth user metadata: `supabase.auth.admin.updateUserById(request.userId, { user_metadata: { user_type: 'resident' } })`. Note: this requires the service role key. Use `createClient` from `@supabase/supabase-js` directly with `SUPABASE_SERVICE_ROLE_KEY` for admin operations. Create an inline admin client: `import { createClient as createAdminClient } from '@supabase/supabase-js'` then `createAdminClient(url, serviceRoleKey)`.
     - `revalidatePath('/admin/upgrade-requests')`.
   - Export `async function rejectUpgrade(formData: FormData)`:
     - Similar to approve but sets `status = 'rejected'`, `upgradeStatus = 'rejected'` on user. No metadata update needed.
     - `revalidatePath('/admin/upgrade-requests')`.

         { id: SEED_IDS.guestUser, email: "demo@starcity.com", password: "password123",

         { id: SEED_IDS.contentStaff, email: "content@starcity.com", password: "admin123",
           user_metadata: { name: "Ma Su Su", role: "content_manager" } },
         { id: SEED_IDS.supportStaff, email: "support@starcity.com", password: "admin123",
           user_metadata: { name: "Ko Myo", role: "ticket_handler" } },
       ];

       for (const u of authUsers) {

         const { data, error } = await supabaseAdmin.auth.admin.createUser({

           password: u.password,
           email_confirm: true,  // Skip email verification for seed users

         if (error) {

             console.log(`  Auth user ${u.email} already exists, updating...`);

             const { data: { users } } = await supabaseAdmin.auth.admin.listUsers();

             if (existing) {
               await supabaseAdmin.auth.admin.updateUserById(existing.id, {
                 password: u.password,
                 user_metadata: u.user_metadata,
                 email_confirm: true,
               });
             }
