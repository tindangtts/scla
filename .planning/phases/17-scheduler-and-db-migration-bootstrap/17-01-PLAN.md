<objective>
Auto-apply PostgreSQL sequences at server startup so booking and ticket number generation never fails on a fresh deploy.

Purpose: The sequences booking_number_seq and ticket_number_seq are required by the booking/ticket number generation logic (QUAL-02, Phase 15). On a fresh database they do not exist, causing runtime failures. Running the migration SQL at startup with CREATE SEQUENCE IF NOT EXISTS makes this idempotent and deploy-safe.
Output: migration-runner.ts module + index.ts wired to call it before listening.
</objective>

```typescript
import app from "./app.js";
import { logger } from "./lib/logger.js";
import { applyMigrations } from "./lib/migration-runner.js";

const rawPort = process.env["PORT"];

if (!rawPort) {
  throw new Error("PORT environment variable is required but was not provided.");
}

const port = Number(rawPort);

if (Number.isNaN(port) || port <= 0) {
  throw new Error(`Invalid PORT value: "${rawPort}"`);
}

applyMigrations()
  .then(() => {
    app.listen(port, (err) => {
      if (err) {
        logger.error({ err }, "Error listening on port");
        process.exit(1);
      }
      logger.info({ port }, "Server listening");
    });
  })
  .catch((err) => {
    logger.error({ err }, "Failed to apply DB migrations — aborting startup");
    process.exit(1);
  });
```
